Agricultural Sector: Cost-Effective Network Design

Algorithm: Kruskal's Algorithm

This algorithm is ideal for designing a network to connect farms or warehouses with minimum cost.

Explanation:

  • Kruskal's Algorithm creates a Minimum Spanning Tree (MST) by selecting the smallest edges while avoiding cycles.
  • It sorts all edges and uses a Union-Find data structure to check for cycles.

                using namespace std;
                
                struct Edge {
                    int u, v, weight;
                };
                
                bool compareEdges(const Edge &e1, const Edge &e2) {
                    return e1.weight < e2.weight;
                }
                
                class DisjointSet {
                public:
                    vector parent, rank;
                
                    DisjointSet(int n) {
                        parent.resize(n);
                        rank.resize(n, 0);
                        for (int i = 0; i < n; i++)
                            parent[i] = i;
                    }
                
                    int find(int x) {
                        if (parent[x] != x)
                            parent[x] = find(parent[x]);
                        return parent[x];
                    }
                
                    void unionSets(int x, int y) {
                        int rootX = find(x);
                        int rootY = find(y);
                        if (rootX != rootY) {
                            if (rank[rootX] > rank[rootY]) {
                                parent[rootY] = rootX;
                            } else if (rank[rootX] < rank[rootY]) {
                                parent[rootX] = rootY;
                            } else {
                                parent[rootY] = rootX;
                                rank[rootX]++;
                            }
                        }
                    }
                };
                
                vector kruskal(int n, vector &edges) {
                    sort(edges.begin(), edges.end(), compareEdges);
                    DisjointSet ds(n);
                    vector mst;
                    int ecounter = 0;
                    for (const Edge &edge : edges) {
                        if (ecounter == n - 1)
                            break;
                        if (ds.find(edge.u) != ds.find(edge.v)) {
                            mst.push_back(edge);
                            ds.unionSets(edge.u, edge.v);
                            ecounter++;
                        }
                    }
                    return mst;
                }
                
                int main() {
                    int n, m;
                    cin >> n >> m;
                    vector edges(m);
                    for (int i = 0; i < m; i++) {
                        cin >> edges[i].u >> edges[i].v >> edges[i].weight;
                    }
                    vector mst = kruskal(n, edges);
                    for (const Edge &edge : mst) {
                        cout << edge.u << " " << edge.v << " " << edge.weight << endl;
                    }
                    return 0;
                }
